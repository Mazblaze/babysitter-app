        // Carica dati (Firebase o locale)

        async function loadData() {

            updateSyncStatus('loading');


            if (isFirebaseEnabled && familyCode) {

                try {

                    const snapshot = await database.ref('families/' + familyCode).once('value');

                    const data = snapshot.val();

                    

                    if (data) {

                        localData = {

                            totalHours: data.totalHours || 0,

                            hourlyRate: data.hourlyRate || 10.0,

                            entries: data.entries || [],

                            payments: data.payments || []

                        };

                        updateSyncStatus('online');

                    } else {

                        // Prima volta, crea i dati iniziali

                        await saveData();

                        updateSyncStatus('online');

                    }

                } catch (error) {

                    console.error('Errore caricamento Firebase:', error);

                    loadLocalData();

                    updateSyncStatus('offline');

                }

            } else {

                loadLocalData();

                updateSyncStatus('offline');

            }


            updateDisplay();

        }


        // Carica dati locali

        function loadLocalData() {

            const saved = localStorage.getItem('babysitter_data_' + familyCode);

            if (saved) {

                localData = JSON.parse(saved);

            }

        }


        // Salva dati (Firebase o locale)

        async function saveData() {

            // Salva sempre localmente

            localStorage.setItem('babysitter_data_' + familyCode, JSON.stringify(localData));


            // Salva su Firebase se disponibile

            if (isFirebaseEnabled && familyCode) {

                try {

                    await database.ref('families/' + familyCode).set({

                        ...localData,

                        lastUpdate: firebase.database.ServerValue.TIMESTAMP,

                        lastUser: currentUser

                    });

                    updateSyncStatus('online');

                    return true;

                } catch (error) {

                    console.error('Errore salvataggio Firebase:', error);

                    updateSyncStatus('offline');

                    return false;

                }

            }

            return false;

        }


        // Aggiorna visualizzazione

        function updateDisplay() {

            document.getElementById('totalHours').textContent = localData.totalHours.toFixed(2) + 'h';

            document.getElementById('totalAmount').textContent = '€' + (localData.totalHours * localData.hourlyRate).toFixed(2);

            document.getElementById('hourlyRate').textContent = localData.hourlyRate.toFixed(2);

            

            const lastEntry = localData.entries[localData.entries.length - 1];

            if (lastEntry) {

                const date = new Date(lastEntry.date).toLocaleString('it-IT');

                document.getElementById('lastUpdate').textContent = `Ultimo aggiornamento: ${date}`;

            }

        }


        // Aggiorna stato sincronizzazione

        function updateSyncStatus(status) {

            const statusEl = document.getElementById('syncStatus');

            switch(status) {

                case 'online':

                    statusEl.textContent = '🌐 Sincronizzato';

                    statusEl.className = 'sync-status';

                    break;

                case 'offline':

                    statusEl.textContent = '📱 Solo locale';

                    statusEl.className = 'sync-status offline';

                    break;

                case 'loading':

                    statusEl.textContent = '⏳ Caricamento...';

                    statusEl.className = 'sync-status offline';

                    break;

            }

        }

            const historyList = document.getElementById('historyList');

            historyList.innerHTML = '';

            

            // Mostra le ultime entrate

            if (localData.entries.length > 0) {

                historyList.innerHTML += '<h4 style="margin-bottom: 10px;">🕐 Ore Correnti:</h4>';

                localData.entries.slice(-15).reverse().forEach(entry => {

                    const date = new Date(entry.date).toLocaleDateString('it-IT');

                    const time = new Date(entry.date).toLocaleTimeString('it-IT', {hour: '2-digit', minute: '2-digit'});

                    const note = entry.note ? ` - ${entry.note}` : '';

                    historyList.innerHTML += `

                        <div class="history-item">

                            <strong>${entry.hours}h</strong>${note}

                            <div class="history-meta">📅 ${date} ${time} | 👤 ${entry.user || 'Sconosciuto'}</div>

                        </div>

                    `;

                });

            }

            

            // Mostra gli ultimi pagamenti

            if (localData.payments.length > 0) {

                historyList.innerHTML += '<h4 style="margin: 20px 0 10px 0;">💰 Ultimi Pagamenti:</h4>';

                localData.payments.slice(-8).reverse().forEach(payment => {

                    const date = new Date(payment.date).toLocaleDateString('it-IT');

                    historyList.innerHTML += `

                        <div class="history-item">

                            <strong>€${payment.amount.toFixed(2)}</strong> (${payment.hours.toFixed(2)}h)

                            <div class="history-meta">📅 ${date} | 👤 ${payment.user || 'Sconosciuto'}</div>

                        </div>

                    `;

                });

            }

            

            if (localData.entries.length === 0 && localData.payments.length === 0) {

                historyList.innerHTML = '<p style="text-align: center; opacity: 0.7; padding: 40px;">Nessun dato ancora</p>';

            }

        }


        function hideAllSections() {

            document.getElementById('mainMenu').classList.add('hidden');

            document.getElementById('addHoursSection').classList.add('hidden');

            document.getElementById('setRateSection').classList.add('hidden');

            document.getElementById('historySection').classList.add('hidden');

        }


        // Aggiungi ore

        async function addHours() {

            const hoursInput = document.getElementById('hoursInput');

            const noteInput = document.getElementById('noteInput');

            

            const hours = parseFloat(hoursInput.value);

            const note = noteInput.value.trim();

            

            if (!hours || hours <= 0 || hours > 24) {

                showMessage('Inserisci un numero di ore valido (0.25 - 24)', 'error');

                return;

            }

            

            // Aggiungi l'entrata

            const entry = {

                date: new Date().toISOString(),

                hours: hours,

                note: note || null,

                user: currentUser

            };

            

            localData.entries.push(entry);

            localData.totalHours += hours;

            

            const saved = await saveData();

            updateDisplay();

            

            const syncText = saved ? ' (sincronizzato)' : ' (salvato localmente)';

            showMessage(`✅ Aggiunte ${hours}h${note ? ' - ' + note : ''}!${syncText}`, 'success');

            

            // Reset form

            hoursInput.value = '';

            noteInput.value = '';

            

            setTimeout(showMainMenu, 2000);

        }


        // Segna come pagato

        async function markAsPaid() {

            if (localData.totalHours === 0) {

                showMessage('Non ci sono ore da pagare!', 'error');

                return;

            }

            

            const amount = localData.totalHours * localData.hourlyRate;

            

            // Salva il pagamento

            localData.payments.push({

                date: new Date().toISOString(),

                hours: localData.totalHours,

                amount: amount,

                rate: localData.hourlyRate,

                user: currentUser

            });

            

            // Azzera il contatore

            localData.totalHours = 0;

            localData.entries = [];

            

            const saved = await saveData();

            updateDisplay();

            

            const syncText = saved ? ' (sincronizzato)' : ' (salvato localmente)';

            showMessage(`💰 Pagamento di €${amount.toFixed(2)} registrato! Contatore azzerato.${syncText}`, 'success');

        }


        // Imposta nuova tariffa

        async function setRate() {

            const rateInput = document.getElementById('rateInput');

            const newRate = parseFloat(rateInput.value);

            

            if (!newRate || newRate <= 0) {

                showMessage('Inserisci una tariffa valida', 'error');

                return;

            }

            

            localData.hourlyRate = newRate;

            const saved = await saveData();

            updateDisplay();

            

            const syncText = saved ? ' (sincronizzato)' : ' (salvato localmente)';

            showMessage(`⚙️ Tariffa aggiornata a €${newRate.toFixed(2)}/ora!${syncText}`, 'success');

            

            setTimeout(showMainMenu, 2000);

        }

        // Mostra messaggio

        function showMessage(message, type = 'success') {

            const messageArea = document.getElementById('messageArea');

            const messageEl = document.createElement('div');

            messageEl.className = `message ${type}-message`;

            messageEl.textContent = message;

            

            messageArea.innerHTML = '';

            messageArea.appendChild(messageEl);

            

            setTimeout(() => {

                messageEl.remove();

            }, 4000);

        }
