    <script>

        // Dati dell'app salvati localmente

        let currentUser = '';

        let appData = {

            totalHours: 0,

            hourlyRate: 10.0,

            entries: [],

            payments: []

        };


        // Carica i dati salvati nel browser

        function loadData() {

            try {

                const savedUser = localStorage.getItem('babysitter_current_user');

                const savedData = localStorage.getItem('babysitter_app_data');

                

                if (savedUser) {

                    currentUser = savedUser;

                }

                

                if (savedData) {

                    appData = JSON.parse(savedData);

                }

                

                updateDisplay();

            } catch (error) {

                console.error('Errore caricamento dati:', error);

                // Se ci sono errori, usa i dati di default

                appData = {

                    totalHours: 0,

                    hourlyRate: 10.0,

                    entries: [],

                    payments: []

                };

            }

        }


        // Salva i dati nel browser

        function saveData() {

            try {

                localStorage.setItem('babysitter_current_user', currentUser);

                localStorage.setItem('babysitter_app_data', JSON.stringify(appData));

                return true;

            } catch (error) {

                console.error('Errore salvataggio dati:', error);

                showMessage('❌ Errore nel salvataggio dati', 'error');

                return false;

            }

        }


        // Setup utente iniziale

        function setupUser() {

            const userName = document.getElementById('userNameInput').value;

            const customName = document.getElementById('customNameInput').value;


            if (!userName) {

                showMessage('❌ Seleziona il tuo nome!', 'error');

                return;

            }


            currentUser = userName === 'Altro' ? customName : userName;


            if (!currentUser) {

                showMessage('❌ Inserisci il nome personalizzato!', 'error');

                return;

            }


            document.getElementById('currentUser').textContent = currentUser;

            document.getElementById('setupSection').classList.add('hidden');

            document.getElementById('appSection').classList.remove('hidden');


            saveData();

            loadData();

            showMessage(`👋 Benvenuto ${currentUser}!`, 'success');

        }


        // Reset setup

        function resetSetup() {

            currentUser = '';

            localStorage.removeItem('babysitter_current_user');

            document.getElementById('setupSection').classList.remove('hidden');

            document.getElementById('appSection').classList.add('hidden');

            document.getElementById('userNameInput').value = '';

            document.getElementById('customNameInput').value = '';

            document.getElementById('customNameGroup').style.display = 'none';

        }


        // Gestione nome personalizzato

        document.getElementById('userNameInput').addEventListener('change', function() {

            const customGroup = document.getElementById('customNameGroup');

            if (this.value === 'Altro') {

                customGroup.style.display = 'block';

                document.getElementById('customNameInput').focus();

            } else {

                customGroup.style.display = 'none';

            }

        });


        // Aggiorna la visualizzazione

        function updateDisplay() {

            document.getElementById('totalHours').textContent = appData.totalHours.toFixed(2) + 'h';

            document.getElementById('totalAmount').textContent = '€' + (appData.totalHours * appData.hourlyRate).toFixed(2);

            document.getElementById('hourlyRate').textContent = appData.hourlyRate.toFixed(2);

            

            if (appData.entries.length > 0) {

                const lastEntry = appData.entries[appData.entries.length - 1];

                const date = new Date(lastEntry.date).toLocaleString('it-IT');

                document.getElementById('lastUpdate').textContent = `Ultimo: ${date} da ${lastEntry.user}`;

            } else {

                document.getElementById('lastUpdate').textContent = 'Nessun inserimento ancora';

            }

        }

        function showSetRate() {

            hideAllSections();

            document.getElementById('rateInput').value = appData.hourlyRate;

            document.getElementById('rateInput').focus();

        }


        function showHistory() {

            hideAllSections();

            document.getElementById('historySection').classList.remove('hidden');

            

            const historyList = document.getElementById('historyList');

            historyList.innerHTML = '';

            

            // Mostra le ore correnti (non pagate)

            if (appData.entries.length > 0) {

                historyList.innerHTML += '<h4 style="margin-bottom: 10px; color: #FFD700;">🕐 Ore da Pagare:</h4>';

                appData.entries.slice(-15).reverse().forEach(entry => {

                    const date = new Date(entry.date).toLocaleDateString('it-IT');

                    const time = new Date(entry.date).toLocaleTimeString('it-IT', {hour: '2-digit', minute: '2-digit'});

                    const note = entry.note ? ` - ${entry.note}` : '';

                    historyList.innerHTML += `

                        <div class="history-item">

                            <strong>${entry.hours}h</strong>${note}

                            <div class="history-meta">📅 ${date} ${time} | 👤 ${entry.user || 'Sconosciuto'}</div>

                        </div>

                    `;

                });

            }

            

            // Mostra i pagamenti passati

            if (appData.payments.length > 0) {

                historyList.innerHTML += '<h4 style="margin: 20px 0 10px 0; color: #51cf66;">💰 Pagamenti Effettuati:</h4>';

                appData.payments.slice(-8).reverse().forEach(payment => {

                    const date = new Date(payment.date).toLocaleDateString('it-IT');

                    historyList.innerHTML += `

                        <div class="history-item">

                            <strong>€${payment.amount.toFixed(2)}</strong> (${payment.hours.toFixed(2)}h)

                            <div class="history-meta">📅 ${date} | 👤 ${payment.user || 'Sconosciuto'}</div>

                        </div>

                    `;

                });

            }

            

            if (appData.entries.length === 0 && appData.payments.length === 0) {

                historyList.innerHTML = '<p style="text-align: center; opacity: 0.7; padding: 40px;">Nessun dato ancora<br><br>Inizia aggiungendo le prime ore! 🚀</p>';

            }

        }


        function hideAllSections() {

            document.getElementById('mainMenu').classList.add('hidden');

            document.getElementById('addHoursSection').classList.add('hidden');

            document.getElementById('setRateSection').classList.add('hidden');

            document.getElementById('historySection').classList.add('hidden');

        }


        // Aggiungi ore

        function addHours() {

            const hoursInput = document.getElementById('hoursInput');

            const noteInput = document.getElementById('noteInput');

            

            const hours = parseFloat(hoursInput.value);

            const note = noteInput.value.trim();

            

            if (!hours || hours <= 0 || hours > 24) {

                showMessage('❌ Inserisci un numero di ore valido (0.25 - 24)', 'error');

                hoursInput.focus();

                return;

            }

            

            // Crea la nuova entry

            const entry = {

                date: new Date().toISOString(),

                hours: hours,

                note: note || null,

                user: currentUser

            };

            

            // Aggiungi ai dati

            appData.entries.push(entry);

            appData.totalHours += hours;

            

            // Salva e aggiorna

            saveData();

            updateDisplay();

            

            showMessage(`✅ Aggiunte ${hours}h${note ? ' - ' + note : ''}!`, 'success');

            

            // Reset del form

            hoursInput.value = '';

            noteInput.value = '';

            

            // Torna al menu dopo 2 secondi

            setTimeout(showMainMenu, 2000);

        }


        // Segna come pagato

        function markAsPaid() {

            if (appData.totalHours === 0) {

                showMessage('❌ Non ci sono ore da pagare!', 'error');

                return;

            }

            

            const amount = appData.totalHours * appData.hourlyRate;

            

            // Conferma prima di procedere

            if (!confirm(`Confermi il pagamento di €${amount.toFixed(2)} per ${appData.totalHours.toFixed(2)} ore?`)) {

                return;

            }

            

            // Salva il pagamento

            appData.payments.push({

                date: new Date().toISOString(),

                hours: appData.totalHours,

                amount: amount,

                rate: appData.hourlyRate,

                user: currentUser

            });

            

            // Azzera il contatore

            appData.totalHours = 0;

            appData.entries = [];

            

            // Salva e aggiorna

            saveData();

            updateDisplay();

            

            showMessage(`💰 Pagamento di €${amount.toFixed(2)} registrato! Contatore azzerato.`, 'success');

        }


        // Cambia tariffa

        function setRate() {

            const rateInput = document.getElementById('rateInput');

            const newRate = parseFloat(rateInput.value);

            

            if (!newRate || newRate <= 0 || newRate > 100) {

                showMessage('❌ Inserisci una tariffa valida (1 - 100 €/ora)', 'error');

                rateInput.focus();

                return;

            }

            

            const oldRate = appData.hourlyRate;

            appData.hourlyRate = newRate;

            

            saveData();

            updateDisplay();

            

            showMessage(`⚙️ Tariffa cambiata da €${oldRate.toFixed(2)} a €${newRate.toFixed(2)}/ora!`, 'success');

            

            setTimeout(showMainMenu, 2000);

        }


        // Mostra messaggio

        function showMessage(message, type = 'success') {

            const messageArea = document.getElementById('messageArea');

            const messageEl = document.createElement('div');

            messageEl.className = `message ${type}-message`;

            messageEl.textContent = message;

            

            // Rimuovi messaggi precedenti

            messageArea.innerHTML = '';

            messageArea.appendChild(messageEl);

            

            // Rimuovi dopo 4 secondi

            setTimeout(() => {

                messageEl.remove();

            }, 4000);

        }
